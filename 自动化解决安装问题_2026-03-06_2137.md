### 5. 完整自动化脚本

```powershell
# 自动化解决 GitHub 克隆问题

function Test-GitHubConnection {
    $connection = Test-NetConnection -ComputerName github.com -Port 443
    return $connection.TcpTestSucceeded
}

function Disable-VPN {
    $vpn = Get-NetAdapter | Where-Object {$_.InterfaceDescription -like "*VPN*"}
    if ($vpn) {
        Write-Host "检测到 VPN，尝试禁用..."
        Disable-NetAdapter -Name $vpn.Name -Confirm:$false
        return $true
    }
    return $false
}

function Clone-Repository {
    param(
        [string]$repoUrl,
        [string]$repoName
    )
    
    # 尝试 HTTPS 克隆
    try {
        git clone $repoUrl --depth 1
        Write-Host "$repoName HTTPS 克隆成功"
        return $true
    } catch {
        Write-Host "$repoName HTTPS 克隆失败"
    }
    
    # 尝试 SSH 克隆
    try {
        $sshUrl = $repoUrl -replace "https://github.com/", "git@github.com:"
        git clone $sshUrl
        Write-Host "$repoName SSH 克隆成功"
        return $true
    } catch {
        Write-Host "$repoName SSH 克隆失败"
    }
    
    # 尝试手动下载
    try {
        $zipUrl = $repoUrl -replace ".git$", "/archive/refs/heads/main.zip"
        $zipFile = "$repoName.zip"
        Invoke-WebRequest -Uri $zipUrl -OutFile $zipFile
        Expand-Archive -Path $zipFile -DestinationPath "."
        Write-Host "$repoName 手动下载成功"
        return $true
    } catch {
        Write-Host "$repoName 手动下载失败"
    }
    
    return $false
}

# 主程序
Write-Host "开始自动化解决 GitHub 克隆问题..."

# 步骤 1: 禁用 VPN
Disable-VPN

# 步骤 2: 测试 GitHub 连接
if (Test-GitHubConnection) {
    Write-Host "GitHub 连接正常"
} else {
    Write-Host "GitHub 连接失败"
}

# 步骤 3: 克隆仓库
$repos = @(
    @{Url="https://github.com/openakita/openakita.git"; Name="OpenAkita"},
    @{Url="https://github.com/eclaire-labs/eclaire.git"; Name="Eclaire"},
    @{Url="https://github.com/Vrooli/Vrooli.git"; Name="Vrooli"}
)

foreach ($repo in $repos) {
    Write-Host "正在克隆 $($repo.Name)..."
    Clone-Repository -repoUrl $repo.Url -repoName $repo.Name
}

Write-Host "自动化解决完成"
```
